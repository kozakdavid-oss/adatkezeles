Tábla létrehozása CREATE TABLE IF NOT EXISTS elfogadasok ( id BIGSERIAL PRIMARY KEY, nev TEXT NOT NULL, anyjaneve TEXT NOT NULL, szuletesi_hely TEXT NOT NULL, szuletesi_ido DATE NOT NULL, szemelyi_ig_szam TEXT NOT NULL, email TEXT NOT NULL, telefon TEXT NOT NULL, cegnev TEXT, lakcim TEXT NOT NULL, adoszam TEXT, szerepkor TEXT, tajékoztato_verzio TEXT NOT NULL DEFAULT '2025/1', ip_cim TEXT, user_agent TEXT, elfogadva_at TIMESTAMPTZ NOT NULL DEFAULT NOW(), created_at TIMESTAMPTZ NOT NULL DEFAULT NOW() ); -- 2. Komment a táblához COMMENT ON TABLE elfogadasok IS 'Adatkezelési tájékoztató elfogadások – időbélyeggel, IP-vel rögzítve'; -- 3. RLS (Row Level Security) bekapcsolása ALTER TABLE elfogadasok ENABLE ROW LEVEL SECURITY; -- 4. Policy: csak INSERT engedélyezett az anon usernek (olvasni nem tud) CREATE POLICY "Bárki beilleszthet" ON elfogadasok FOR INSERT TO anon WITH CHECK (true); -- 5. Policy: authenticated user (te) mindent láthat CREATE POLICY "Admin mindent lát" ON elfogadasok FOR SELECT TO authenticated USING (true); -- 6. Index az email-re (gyors kereséshez) CREATE INDEX idx_elfogadasok_email ON elfogadasok(email); CREATE INDEX idx_elfogadasok_datum ON elfogadasok(elfogadva_at DESC); -- ================================================================ -- KÉSZ – ezután a supabase_edge_function.ts fájlt deploy-old -- ================================================================
